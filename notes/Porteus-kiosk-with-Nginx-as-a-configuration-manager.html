Porteus kiosk with nginx as a configuration manager

Porteus Kiosk - free and open source kiosk software for web terminals.

Porteus checks for remote configs on startup, and reimages the OS if a configuration change is detected.

server {
  listen      80;
  server_name sub.domain.tld;
  return      302 https://$server_name$request_uri;
}

server {
  listen      443 ssl;
  server_name sub.domain.tld;

  # Porteus supports both application/octet-stream and text/plain,
  # but for simplisity's sake we'll use text/plain, so you can see
  # the config directly from a browser.
  default_type            text/plain; # text/plain || application/octet-stream;
  add_header Content-Type text/plain; # text/plain || application/octet-stream;

  # Location(s) that don't require authentication
  location /policies/firefox.txt {
    # Available parameters can be found in Firefox -> about:config
    set $CFG '';
    set $CFG '${CFG}user_pref("gfx.webrender.all", true);\n';
    set $CFG '${CFG}user_pref("svg.context-properties.content.enabled", true);\n';
    set $CFG '${CFG}user_pref("layers.acceleration.force-enabled", true);\n';
    # To set the homepage using Firefox config:
    # set $CFG '${CFG}user_pref("browser.startup.homepage", "https://google.com");\n';

    return 200 $CFG;
  }

  # ref. https://stackoverflow.com/a/67981564 - https://nginx.org/en/docs/dev/development_guide.html#http_phases
  # `return` (`NGX_HTTP_SERVER_REWRITE_PHASE`) is processed before `auth_basic*` (`NGX_HTTP_ACCESS_PHASE`) in nginx,
  # meaning you can't use basic auth to directly authenticate a `return`-statement (in the same location-block).
  # `auth_basic*` precedes `proxy_pass` and will therefore work as a semi-hacky workaround.
  # `sudo htpasswd -c /etc/nginx/basic_auth kiosk`
  location / {
    auth_basic           "Kiosk Config Authentication";
    auth_basic_user_file /etc/nginx/basic_auth;
    proxy_pass           http://127.0.0.1:1337;
  }
}

server {
  listen 1337;

  # Full list of configuration-parameters: https://porteus-kiosk.org/parameters.html
  # Information about remote-management: https://porteus-kiosk.org/remote-management.html

  location = /kiosks/kiosk-a.txt {
    set $CFG '';
    set $CFG '${CFG}connection=wifi\n'; # wifi/wired
    set $CFG '${CFG}dhcp=yes\n';
    # set $CFG '${CFG}network_interface=eth0\n';
    # set $CFG '${CFG}dns_server=1.1.1.1 1.0.0.1\n';
    # set $CFG '${CFG}default_gateway=172.16.0.1\n';
    # set $CFG '${CFG}ip_address=172.16.0.64\n';
    # set $CFG '${CFG}netmask=255.255.255.0\n';
    set $CFG '${CFG}wifi_encryption=wpa\n';
    set $CFG '${CFG}wpa_password=WIFI_PASSWORD\n';
    set $CFG '${CFG}ssid_name=WIFI_SSID\n';
    set $CFG '${CFG}browser=firefox\n';
    set $CFG '${CFG}homepage=https://duck.com\n';
    set $CFG '${CFG}hostname=kiosk\n';
    set $CFG '${CFG}hardware_video_decode=yes\n';
    set $CFG '${CFG}allow_icmp_protocol=yes\n';
    set $CFG '${CFG}primary_keyboard_layout=no\n';
    set $CFG '${CFG}hide_mouse=5\n';
    set $CFG '${CFG}disable_navigation_bar=yes\n';

    set $CFG '${CFG}vnc_password=vncpass\n';
    set $CFG '${CFG}additional_components=09-x11vnc.xzm uefi.zip\n';

    return 200 $CFG;
  }
